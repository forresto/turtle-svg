var drawRegular = function(sideCount, sideLength) {
  for (var i=0; i<sideCount; i++){
    moveForward(sideLength);
    turnRight(1/sideCount);
  }
};
drawRegular(5, 20);

// Draw a recursive space-filling curve:
var segmentLength = 15;
function unit(iteration) {
  if (iteration > 0) {
    unit(iteration - 1);
    r(0.25);
    unit(iteration - 1);
    f(segmentLength);
    unit(iteration - 1);
    r(0.25);
    unit(iteration - 1);
  }
}
moveTo(30, 20);
turnTo(3/8);
unit(6);
f(segmentLength);
unit(6);

// Draw golden ratio squares with Fibonacci
var side = 3;
var lastSide = 0;
var nextSide;
moveTo(550, 350);
while( side < 1200 ){
    // Draw square
    f(side);
    l(1/4);
    f(side);
    l(1/4);
    f(side);
    l(1/4);
    f(side);
    l(1/4);
    f(side);
    l(1/4);
    f(side);
    // Advance fib
    nextSide = side + lastSide;
    lastSide = side;
    side = nextSide;
}
